<br>

```plsql
CREATE TABLE MARLOC(
    MARNAME VARCHAR2(50) PRIMARY KEY NOT NULL,
    MARNO NUMBER NUMBER(15) UNIQUE NOT NULL,
    LAT NUMBER(15,15) NOT NULL,
    LNG NUMBER(15,15) NOT NULL,
    EXPL VARCHAR2(200)	
);

CREATE SEQUENCE SEQ_NNO;
COMMIT;

```
