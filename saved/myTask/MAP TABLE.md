  STARTDAY DATE,
  ENDDAY DATE,
  URL VARCHAR2(300),
  COLOR VARCHAR2(50) DEFAULT '#FFEEDD' NOT NULL,
  COLORTEXT VARCHAR2(50) DEFAULT 'BLACK' NOT NULL,
  ATTACHEDFILE VARCHAR2(300) DEFAULT 'NULL'
